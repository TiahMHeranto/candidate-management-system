Backend dependences:

# Core
npm install express mongoose dotenv cors

# Auth & security
npm install jsonwebtoken bcrypt express-rate-limit

# Validation
npm install zod

# Logs
npm install pino pino-http

# Dev
npm install -D typescript ts-node @types/node @types/express @types/jsonwebtoken
